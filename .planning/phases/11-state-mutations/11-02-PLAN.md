    Port `execGit` helper:
    ```typescript
    import { execSync } from 'node:child_process';

    function execGit(cwd: string, args: string[]): { exitCode: number; stdout: string; stderr: string } {
      try {
        const stdout = execSync(`git ${args.join(' ')}`, {
          cwd,
          encoding: 'utf-8',
          stdio: ['pipe', 'pipe', 'pipe'],
        }).trim();
        return { exitCode: 0, stdout, stderr: '' };
      } catch (err: unknown) {
        if (err instanceof Error && 'status' in err) {
          const execErr = err as { status: number; stdout?: string; stderr?: string };
          return {
            exitCode: execErr.status || 1,
            stdout: String(execErr.stdout || '').trim(),
            stderr: String(execErr.stderr || '').trim(),
          };
        }
        return { exitCode: 1, stdout: '', stderr: String(err) };
      }
    }
    ```

    Port `sanitizeCommitMessage(msg)`: Strips null bytes, control chars (except \n\t), and known injection markers (`<|`, `|>`, `SYSTEM:`, `ASSISTANT:`).
