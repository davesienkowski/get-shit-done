function replaceInCurrentMilestone(content, pattern, replacement) {
  const lastDetailsClose = content.lastIndexOf('</details>');
  if (lastDetailsClose === -1) return content.replace(pattern, replacement);
  const offset = lastDetailsClose + '</details>'.length;
  return content.slice(0, offset) + content.slice(offset).replace(pattern, replacement);
}
